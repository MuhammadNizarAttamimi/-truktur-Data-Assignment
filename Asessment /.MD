#include "SLLInventory.h"
#include <iostream>
#include <iomanip>
using namespace std;

float hargaAkhir(Product p) {
    return p.HargaSatuan * (1 - p.DiskonPersen / 100);
}

void createList(List &L) { L.head = NULL; }
bool isEmpty(List L) { return L.head == NULL; }

Node* allocate(Product P) {
    Node* N = new Node;
    N->info = P;
    N->next = NULL;
    return N;
}
void deallocate(Node* P) { delete P; }

void insertFirst(List &L, Product P) {
    Node* N = allocate(P);
    N->next = L.head;
    L.head = N;
}

void insertLast(List &L, Product P) {
    Node* N = allocate(P);
    if (isEmpty(L)) L.head = N;
    else {
        Node* Q = L.head;
        while (Q->next) Q = Q->next;
        Q->next = N;
    }
}

void insertAfter(List &L, Node* Q, Product P) {
    if (!Q) return;
    Node* N = allocate(P);
    N->next = Q->next;
    Q->next = N;
}

void deleteFirst(List &L, Product &P) {
    if (isEmpty(L)) return;
    Node* N = L.head;
    P = N->info;
    L.head = N->next;
    deallocate(N);
}

void deleteLast(List &L, Product &P) {
    if (isEmpty(L)) return;
    Node *Q = L.head, *prev = NULL;

    while (Q->next) {
        prev = Q;
        Q = Q->next;
    }
    P = Q->info;

    if (!prev) L.head = NULL;
    else prev->next = NULL;
    deallocate(Q);
}

void deleteAfter(List &L, Node* Q, Product &P) {
    if (!Q || !Q->next) return;
    Node* N = Q->next;
    P = N->info;
    Q->next = N->next;
    deallocate(N);
}

void updateAtPosition(List &L, int pos, Product newP) {
    int idx = 1;
    Node* Q = L.head;
    while (Q && idx < pos) {
        Q = Q->next;
        idx++;
    }
    if (Q) Q->info = newP;
}

void viewList(List L) {
    if (isEmpty(L)) {
        cout << "(kosong)\n";
        return;
    }
    int idx = 1;
    Node* Q = L.head;

    while (Q) {
        cout << idx++ << ". "
             << Q->info.Nama << " | "
             << Q->info.SKU << " | "
             << Q->info.Jumlah << " | "
             << Q->info.HargaSatuan << " | "
             << Q->info.DiskonPersen << "% | Akhir="
             << fixed << setprecision(2) << hargaAkhir(Q->info) << endl;
        Q = Q->next;
    }
}

void searchByFinalPriceRange(List L, float minP, float maxP) {
    int idx = 1;
    Node* Q = L.head;
    bool found = false;

    while (Q) {
        float h = hargaAkhir(Q->info);
        if (h >= minP && h <= maxP) {
            found = true;
            cout << idx << ". " << Q->info.Nama 
                 << " hargaAkhir=" << h << endl;
        }
        Q = Q->next;
        idx++;
    }
    if (!found) cout << "Tidak ada produk dalam rentang.\n";
}

void MaxHargaAkhir(List L) {
    if (isEmpty(L)) {
        cout << "(kosong)\n";
        return;
    }
    float maxH = hargaAkhir(L.head->info);
    Node* Q = L.head;

    while (Q) {
        float h = hargaAkhir(Q->info);
        if (h > maxH) maxH = h;
        Q = Q->next;
    }

    cout << "Produk harga akhir tertinggi = " << maxH << endl;
    Q = L.head;
    int idx = 1;

    while (Q) {
        if (hargaAkhir(Q->info) == maxH)
            cout << idx << ". " << Q->info.Nama << endl;
        Q = Q->next;
        idx++;
    }
}
